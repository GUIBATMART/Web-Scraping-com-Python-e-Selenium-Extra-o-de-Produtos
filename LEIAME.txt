# Automação Web e Engenharia de Dados com Linguagem Python e Selenium

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsaselenium python=3.13

# Ative o ambiente:

conda activate dsaselenium (ou: source activate dsaselenium)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Execute os comandos abaixo:

python hello.py
python web_scraping_1.py
python web_scraping_2.py

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name dsaselenium --all
